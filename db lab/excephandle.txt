create or replace procedure minsalary(empno in number)
is
  v_salary number;
  minSal number:=10000;
  myexp exception;
begin
  select salary into v_salary from employee where emp_id=empno;
  if v_salary<minSal then
    raise myexp;
  else 
    dbms_output.put_line('The salary is above minimum');
  end if;
  exception
  when myexp then
    dbms_output.put_line('The salary is less');
end;
/
declare
  empid number:=&empid;
begin
  minsalary(empid);
end;
/
